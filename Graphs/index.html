  • const g = new Graph();
  • g.addVertex("A").addVertex("B").addVertex("C").addVertex("D").addVertex("E").addVertex("F");

  • g.breadthFirstSearchIterative("A") // should return ["A", "B", "C", "D", "E", "F"]
